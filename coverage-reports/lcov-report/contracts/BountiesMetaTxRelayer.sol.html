    bytes memory signature,
    address payable [] memory _issuers,
    address [] memory _approvers,
    string memory _data,
    uint _deadline,
    address _token,
    uint _tokenVersion,
    uint _depositAmount,
    uint _nonce)
    public
    payable
    returns (uint)
    { 
    //@audit-ok - input validation, do we need to check array length here? - is being checked in issueBounty
    //@audit-ok check signature.length as modifier and bail early (is done in get.signature)
 
    bytes32 metaHash = keccak256(abi.encodePacked(address(this),

                                                  _nonce));
    address signer = getSigner(metaHash, signature);
 
    //make sure signer doesn't come back as 0x0

                                         _issuers,
                                         _approvers,
                                         _data,

  }

  function getSigner(
    bytes32 _hash,
    bytes memory _signature)
    internal
    pure
    returns (address)
  {
    bytes32 r;
    bytes32 s;
    uint8 v;
    if (_signature.length != 65) {  //@audit-ok
      return address(0);  //@audit-ok - indicate badsign in retunval instead of using address 0
    }
    assembly {
      r := mload(add(_signature, 32))
      s := mload(add(_signature, 64))
      v := byte(0, mload(add(_signature, 96)))  //@audit-ok
    }
    if (v < 27) {
      v += 27;
    }
    
    if (v != 27 && v != 28) {
      return address(0);  //@audit-ok - explcitly indicate errors with badsig retval?
    } else {
      return ecrecover(keccak256(
        abi.encodePacked("\x19Ethereum Signed Message:\n32", _hash)
      ), v, r, s);
    }
  }
